\begin{frame}
\frametitle{Основные допущения}
\begin{itemize}
\item Массообмен между паром и флегмой эквимолярный
\item В каждом сечении по высоте колонны давления и температуры потоков пара и жидкости близки друг к другу (от сечения к сечению они изменяются)
\item Эффекты диффузии между соседними сечениями пренебрежимо малы
\item Теплота выходящих потоков передается потокам, поступающим в колонну, причем необратимостью этого теплообмена можно пренебречь
\item Поток разделяемой смеси в форме жидкости подается при температуре кипения в то сечение колонны, состав флегмы в котором совпадает с составом этого потока
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Термодинамические балансы для колонны бинарной ректификации}
\begin{itemize}
\item Энергетический баланс
$$
q_{+} - q_{-} + g_{_F} h_{_F} - g_{_F}\varepsilon h_{_D} - g_{_F}(1-\varepsilon) h_{_B} = 0
$$
\item Энтропийный баланс
$$
g_{_F} s_{_F} - g_{_F}\varepsilon s_{_D} - g_{_F}(1-\varepsilon) s_{_B}+ \frac{q_{+}}{T_B} -
\frac{q_{-}}{T_D} + \sigma = 0
$$
\item $\sigma > 0$ - производство энтропии в колонне
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Связь производительности и теплового потока}
\begin{itemize}
\item Химические потенциалы: $\mu_i(T, P, x_i)= \mu_{i0}(P, T)+ RT \ln x_i,\quad i=D,B,F$
\item Обратимая работа разделения: $A_i=-RT_D\Big[x_ilnx_i +(1-x_i)ln(1-x_i)\Big],\quad i=D,B,F$
\item Работа разделения Гиббса: $A_G = A_F-\varepsilon A_D-(1-\varepsilon)A_B$
\item КПД Карно: $\eta_{_{\mbox{\scriptsize K}}}= (1- {T_D}/{T_B})$
\item Итоговая зависимость:
$$
g_F=q\frac{\eta_{_{\mbox{\scriptsize K}}}}{A_G}-\sigma(q,g_F)\frac{T_D}{A_G}
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Вклад теплообмена в необратимость}
\begin{itemize}
\item Потоки теплоты в кубе и дефлегматоре пропорциональны разности температур
$$
q=rV=\beta_{_B}(T_+-T_B)=\beta_{_D}(T_D-T_-)
$$
\item Производство энтропии за счет тепловых процессов в кубе и дефлегматоре  равно
$$
\sigma_q = q\Big[\frac{1}{T_B}-\frac{1}{T_+}+\frac{1}{T_-}-\frac{1}{T_D} \Big]=q^2\Big[\frac{1}{\beta_{_B}T_B T_{+}} +\frac{1}{\beta_{_D}T_D T_{-}}\Big]
$$
\item Здесь $r$ - мольная теплота парообразования, $\beta_i$ - коэффициенты теплопереноса
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Вклад массообмена в необратимость}
\begin{itemize}
\item Закон массопереноса пропорционален движущей силе
$$
g(y, y^0)= k\frac{[ \mu_1(T, y^0)-\mu_1(T, y)]}{T}
$$
\item Оценка снизу производства энтропии за счет процессов массопереноса в колонне
$$
\sigma_g\geq\frac{2(x_D-x_B)q^2}{kr^2}
$$
\item Коэффициент массопереноса $k$ зависит от условий внутри колонны и коэффициентов относительной летучести ($\alpha$) компонентов смеси
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Параметризация зависимости производительности колонны от затрат теплоты с учетом необратимости}
\begin{itemize}
\item Обратимый КПД
$$
b=\frac{T_B - T_D}{T_B A_G}=\frac{\eta_k}{A_G}
$$
\item Коэффициент необратимости
$$
a=\Big[\frac{1}{\beta_{_B}T_B T_{+}} +\frac{1}{\beta_{_D}T_D T_{-}}+\frac{2(x_D-x_B)}{kr^2}\Big]\frac{T_D}{A_G}
$$
\item Верхняя оценка производительности колонны бинарной ректификации
$$
g_F= bq-aq^2
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Свойства зависимости производительности от затрат тепла}
\begin{itemize}
\item Производительность максимальна при $q^0=\frac{b}{2a}$ и достигает значения $g_F^m=\frac{b^2}{4a}$
\item Участок от нуля до значения теплового потока, равного $q^0$, образует рабочую часть области реализуемых режимов колонны; дальнейшее увеличение потока теплоты в силу роста диссипации приводит к уменьшению производительности
\item КПД колонны с учетом необратимости $\eta=\frac{g_F}{q}$ для режима  с предельной производительностью не зависит от необратимых факторов и равен $0,5b$
\item Эффективный коэффициент массопереноса $k$, предполагается известным
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Модель процесса разделения трехкомпонентной смеси}
\begin{itemize}
\item Используются две колонны бинарной ректификации
\item Возможны два порядка разделения
\begin{itemize}
\item \textbf{Прямой}, когда в первой колонне отделяют нулевой компонент, а во второй разделяют первый и второй. Соответствующие ему коэффициенты в параметрическом представлении каждой колонны имеют индекс 1; например, $b_{11}$ - обратимый КПД при прямом порядке разделения для первой колонны
\item \textbf{Непрямой}, когда на первой ступени отделяют второй компонент, а оставшуюся смесь делят во второй колонне; $b_{21}$ - обратимый КПД при непрямом порядке разделения для первой колонны
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Модель процесса разделения трехкомпонентной смеси}
\begin{itemize}
\item Исходная смесь с концентрациями веществ $x_0$, $x_1$, $x_2$, где $x_0$ - концентрация низкокипящего, а $x_2$ - высококипящего
\item Мольная теплота парообразования смеси низкокипящего и среднекипящего определяется как $r_{01} = (r_0x_0 + r_1x_1)/(x_0+x_1)$
\item Каскад колонн характеризуется следующими параметрами ($i$ - выбранный порядок разделения)
\begin{itemize}
\item Коэффициент массопереноса для первой колонны $k_{i1}$;
\item Коэффициент массопереноса для второй колонны $k_{i2}$;
\item Коэффициенты теплопереноса для первой колонны, в кубе и дефлегматоре, соответственно $\beta^B_{i1}$, $\beta^D_{i1}$;
\item Коэффициенты теплопереноса для второй колонны $\beta^B_{i2}$, $\beta^D_{i2}$;
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Прямой порядок разделения}
\begin{itemize}
\item Параметры первой колонны:
$$
b_{11} = -\frac{T_1 - T_0}{RT_0T_1\left[x_0\ln{x_0} + (1-x_0)\ln{(1-x_0)}\right]}
$$
$$
a_{11} = -\left[\frac{1}{\beta^B_{11}T_1^2} + \frac{1}{\beta^D_{11}T_0^2} + \frac{2}{k_{11}r_0^2}\right]\frac{1}{R\left[x_0\ln{x_0} + (1-x_0)\ln{(1-x_0)}\right]}
$$
\item Параметры второй колонны:
$$
b_{12} = -\frac{T_2 - T_1}{RT_1T_2(x_1\ln{x_1} + x_2\ln{x_2})}
$$
$$
a_{12} = -\left[\frac{1}{\beta^B_{12}T_2^2} + \frac{1}{\beta^D_{12}T_1^2} + \frac{2}{k_{12}r_1^2}\right]\frac{1}{R(x_0\ln{x_0} + x_2\ln{x_2})}
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Непрямой порядок разделения}
\begin{itemize}
\item Параметры первой колонны:
$$
b_{21} = -\frac{T_2 - T_1}{RT_2T_1\left[(x_0+x_1)\ln{(x_0+x_1)} + x_2\ln{x_2}\right]}
$$
$$
a_{21} = -\left[\frac{1}{\beta^B_{21}T_2^2} + \frac{1}{\beta^D_{21}T_1^2} + \frac{2}{k_{21}r_{01}^2}\right]\frac{1}{R\left[(x_0+x_1)\ln{(x_0+x_1)} + x_2\ln{x_2}\right]}
$$
\item Параметры второй колонны:
$$
b_{22} = -\frac{T_1 - T_0}{RT_1T_0(x_0\ln{x_0} + x_1\ln{x_1})}
$$
$$
a_{22} = -\left[\frac{1}{\beta^B_{22}T_1^2} + \frac{1}{\beta^D_{22}T_0^2} + \frac{2}{k_{22}r_0^2}\right]\frac{1}{R(x_0\ln{x_0} + x_1\ln{x_1})}
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Максимальная производительность каскада}
\begin{itemize}
\item Максимальная производительность каскада  равна максимальной производительности первой колонны
$$
g_F^* = \frac{b_{i1}^2}{4a_{i1}}
$$
\item В качестве первой целесообразно выбрать колонну, для которой максимальная производительность больше (значение коэффициента необратимости меньше), так как на нее поступает весь поток смеси, а обратимый КПД не зависит от кинетики
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Условия согласованности}
\begin{itemize}
\item Колонны должны быть согласованы таким образом, чтобы максимальной производительности первой колонны по исходной смеси, соответствовала допустимая производительность второй по поступающему на нее двухкомпонентному потоку
\item Для прямого порядка
$$
\frac{b_{12}^2}{(1-x_0)a_{12}} \geq \frac{b_{11}^2}{a_{11}}
$$
\item Для непрямого порядка
$$
\frac{b_{22}^2}{(1-x_2)a_{22}} \geq \frac{b_{21}^2}{a_{21}}
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Условия полной согласованности}
\begin{itemize}
\item В оптимальном случае максимальной производительности первой колонны по исходной смеси должна соответствовать максимальная производительность второй колонны по поступающему на нее потоку двухкомпонентной смеси
\item Условия согласованности превращаются в равенства
\item Из условий полной согласованности следует
$$
a_{12} = a_{11}\frac{b_{12}^2}{b_{11}^2(1-x_0)},\quad a_{22} = a_{21}\frac{b_{21}^2}{b_{22}^2(1-x_2)}
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Необратимая оценка производительности каскада в случае полной согласованности}
\begin{itemize}
\item Для прямого порядка
$$
g_F = \frac{b_{11}b_{12}}{b_{12}+b_{11}(1-x_0)}(q_1 + q_2) - \frac{a_{11}b_{12}}{b_{12} + b_{11}(1-x_0)}(q_1 + q_2)^2
$$
\item Для непрямого порядка
$$
g_F = \frac{b_{21}b_{22}}{b_{22}+b_{21}(1-x_2)}(q_1 + q_2) - \frac{a_{21}b_{22}}{b_{22} + b_{21}(1-x_2)}(q_1 + q_2)^2
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Условия превалирования одного из порядков разделения}
\begin{itemize}
\item Прямой порядок разделения заведомо выгоднее непрямого при одновременном соблюдении двух неравенств:
$$
\frac{b_{11}b_{12}}{b_{12}+b_{11}(1-x_0)}>\frac{b_{21}b_{22}}{b_{22}+b_{21}(1-x_2)}
$$
$$
\frac{b_{11}^2b_{12}}{a_{11}\left[b_{12}+b_{11}(1-x_0)\right]} \geq \frac{b_{21}^2b_{22}}{a_{21}\left[b_{22}+b_{21}(1-x_2)\right]}
$$
\item Одно из неравенств обязательно должно быть строгим
\item Первое из неравенств означает, что обратимый КПД для прямого порядка не меньше, чем для непрямого, второе соответствует такому же соотношению для предельных производительностей
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Расчет эффективных значений параметров, определяющих множество реализуемости колонны}
\begin{itemize}
\item Эффективный коэффициент массопереноса
$$
k = \frac{q(x_D - x_B)}{Rr(I_1 - I_2 - I_3)}
$$
{\scriptsize Величины $I$ зависят от производительности, концентраций и коэффициентов относительной летучести $\alpha$; $R$ - газовая постоянная}
\item Коэффициент необратимости и обратимый КПД
$$
a=\frac{q_1 g_{F2}-q_2g_{F1}}{q_1q_2(q_1-q_2)}, \quad b=\frac{g_{F1}}{q_1}+aq_1
$$
\item Флегмовое число
$$
R=\frac{q(g_F)}{g_Fx_Fr}-1=\frac {b - \sqrt{b^2 - 4ag_F}}{2ag_Fx_Fr}-1
$$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Результаты}
\begin{itemize}
\item Показано, что оценка производительности колонны бинарной ректификации, в функции затрат теплоты имеет форму выпуклой вверх параболы и характеризуется двумя характеристическими параметрами: обратимым КПД и коэффициентом необратимости
\item Через эти параметры выражена предельная производительность колонны и КПД, соответствующий предельной производительности
\item Показано, как на значения этих параметров влияют характеристики смеси и кинетические коэффиценты тепло и массопереноса
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Результаты}
\begin{itemize}
\item Предложена процедура уточнения характеристических параметров по данным измерений на действующей колонне
\item Решена задача о выборе последовательности разделения смеси из трех компонентов по условию минимума суммарных затрат теплоты в каскаде из двух колонн
\item  Показано, что на порядок разделения влияет не только кинетика массо- , но и теплообмена в колоннах
\item Предложена  процедура расчета порядка разделения
\end{itemize}
\end{frame}
